473template<typename T>
474std::optional<std::reference_wrapper<const T>> get_ref(Registry& r,
475 JsonObject const& object,
476 std::string const& key)
477{
478 auto it = object.find(key);
479 if (it == object.end()) {
480 return std::nullopt;
481 }
482
483 auto const& value = it->second.value;
484
485 try {
486 std::string hook = std::get<std::string>(value);
487 try {
488 if (hook.starts_with('#')) {
489 std::string striped = hook.substr(1);
490 std::string comp = striped.substr(0, striped.find(':'));
491 std::string value = striped.substr(striped.find(':') + 1);
492 return r.get_hooked_value<T>(comp, value);
493 }
494 } catch (std::bad_any_cast const&) {
495 std::cerr << std::format(
496 R"(Error geting hooked value "{}": Invalid type\n)", hook);
497 return std::nullopt;
498 } catch (std::out_of_range const&) {
499 std::cerr << std::format(
500 R"(Error geting hooked value "{}": Invalid hook\n)", hook);
501 return std::nullopt;
502 }
503 } catch (std::bad_variant_access const&) { // NOLINT intentional fallthrought
504 }
505
506 if constexpr (is_in_json_variant_v<T>) {
507 try {
508 return std::reference_wrapper<const T>(std::get<T>(value));
509 } catch (...) {
510 return std::nullopt;
511 }
512 }
513
514 return std::nullopt;
515}
