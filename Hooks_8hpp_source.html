476template<typename T>
477std::optional<std::reference_wrapper<const T>> get_ref(Registry& r,
478 JsonObject const& object,
479 std::string const& key)
480{
481 if (!object.contains(key)) {
482 return std::nullopt;
483 }
484
485 auto const& value = object.at(key).value;
486
487 const auto* hook = std::get_if<std::string>(&value);
488 if (hook != nullptr) {
489 try {
490 if (hook->starts_with('#')) {
491 std::string striped = hook->substr(1);
492 std::string comp = striped.substr(0, striped.find(':'));
493 std::string value = striped.substr(striped.find(':') + 1);
494 return r.get_hooked_value<T>(comp, value);
495 }
496 } catch (std::bad_any_cast const&) {
497 LOGGER_EVTLESS(
498 LogLevel::ERR,
499 "Hooks",
500 std::format(R"(Error geting hooked value "{}": Invalid type\n)",
501 *hook));
502 return std::nullopt;
503 } catch (std::out_of_range const&) {
504 LOGGER_EVTLESS(
505 LogLevel::ERR,
506 "Hooks",
507 std::format(R"(Error geting hooked value "{}": Invalid hook\n)",
508 *hook));
509 return std::nullopt;
510 }
511 }
512
513 if constexpr (is_in_json_variant_v<T>) {
514 const auto* return_ref = std::get_if<T>(&value);
515 if (return_ref != nullptr) {
516 return std::reference_wrapper<const T>(*return_ref);
517 }
518 }
519
520 return std::nullopt;
521}

536template<typename T>
537std::optional<T> resolve_scoped_hook(Registry& r,
538 std::string const& value_str,
539 Ecs::Entity entity)
540{
541 size_t first_colon = value_str.find(':');
542 size_t second_colon = value_str.find(':', first_colon + 1);
543
544 if (first_colon == std::string::npos || second_colon == std::string::npos) {
545 LOGGER_EVTLESS(
546 LogLevel::ERR,
547 "Hooks",
548 std::format(
549 R"(Error parsing hook: Expected format "scope:component:field", got "{}")",
550 value_str));
551 return std::nullopt;
552 }
553
554 std::string scope = value_str.substr(0, first_colon);
555 std::string comp =
556 value_str.substr(first_colon + 1, second_colon - first_colon - 1);
557 std::string value = value_str.substr(second_colon + 1);
558
559 std::optional<std::reference_wrapper<T>> hooked_val;
560
561 if (scope == "self") {
562 std::string hook_key = comp + "{" + std::to_string(entity) + "}";
563 hooked_val = r.template get_hooked_value<T>(hook_key, value);
564 } else if (scope == "global") {
565 hooked_val = r.template get_global_hooked_value<T>(comp, value);
566 } else {
567 LOGGER_EVTLESS(
568 LogLevel::ERR,
569 "Hooks",
570 std::format(R"(Unknown scope "{}": Expected "self" or "global")",
571 scope));
572 return std::nullopt;
573 }
574
575 if (hooked_val) {
576 return hooked_val->get();
577 }
578
579 return std::nullopt;
580}

696template<typename T, typename... Args>
697std::optional<T> get_value_copy(
698 Registry& r,
699 JsonObject const& object,
700 std::string const& key,
701 std::optional<Ecs::Entity> entity = std::nullopt,
702 Args&&... args)
703{
704 if (!object.contains(key)) {
705 return std::nullopt;
706 }
707
708 if (object.contains(key)) {
709 auto const* value_str = std::get_if<std::string>(&object.at(key).value);
710 if (value_str) {
711 if (value_str->starts_with('@')) {
712 if constexpr (std::is_same_v<T, std::size_t>) {
713 if (value_str->substr(1) == "self") {
714 return entity;
715 }
716 }
717 }
718 if (value_str->starts_with('%')) {
719 std::string stripped = value_str->substr(1);
720 if (entity.has_value()) {
721 auto result = resolve_scoped_hook<T>(r, stripped, entity.value());
722 if (result.has_value()) {
723 return result;
724 }
725 } else {
726 auto result = resolve_scoped_hook<T>(r, stripped);
727 if (result.has_value()) {
728 return result;
729 }
730 }
731 }
732 }
733 }
734 // self hook: @self
735
736 auto tmp = get_ref<T>(r, object, key);
737 if (tmp.has_value()) {
738 return tmp.value().get();
739 }
740
741 if constexpr (std::is_constructible_v<T, JsonObject, Args...>) {
742 if (!object.contains(key)) {
743 return std::nullopt;
744 }
745 const auto* obj = std::get_if<JsonObject>(&object.at(key).value);
746 if (obj != nullptr) {
747 return T(*obj, std::forward<Args>(args)...);
748 }
749 }
750 return std::nullopt;
751}

841template<typename ComponentType, typename T, typename... Args>
842std::optional<T> get_value(Registry& r,
843 JsonObject const& object,
844 Ecs::Entity entity,
845 std::string const& field_name,
846 Args&&... args)
847{
848 if (object.contains(field_name)) {
849 const auto* value_str =
850 std::get_if<std::string>(&object.at(field_name).value);
851
852 if (value_str != nullptr) {
853 // self hook: @self
854 if (value_str->starts_with('@')) {
855 if constexpr (std::is_same_v<T, std::size_t>) {
856 if (value_str->substr(1) == "self") {
857 return entity;
858 }
859 }
860 }
861 // Dynamic hook: #scope:component:field
862 if (value_str->starts_with('#')) {
863 std::string stripped = value_str->substr(1);
864
865 r.template register_binding<ComponentType, T>(
866 entity, field_name, stripped);
867
868 auto result = resolve_scoped_hook<T>(r, stripped, entity);
869 if (result.has_value()) {
870 return result;
871 }
872
873 return T {};
874 }
875
876 if (value_str->starts_with('%')) {
877 std::string stripped = value_str->substr(1);
878
879 auto result = resolve_scoped_hook<T>(r, stripped, entity);
880 if (result.has_value()) {
881 return result;
882 }
883 }
884 }
885 }
886
887 return get_value_copy<T>(
888 r, object, field_name, entity, std::forward<Args>(args)...);
889}
