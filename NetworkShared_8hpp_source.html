396 void push(T const& obj)

399 this->queue.push(obj);
400 this->semaphore.release();
401 }
402
403 T pop()
404 {
405 std::lock_guard<std::mutex> guard(this->lock);
406 auto tmp = this->queue.front();
407 this->queue.pop();

409 }
410
411 std::vector<T> flush()
412 {
413 std::lock_guard<std::mutex> guard(this->lock);
414 std::vector<T> tmp;
415 tmp.reserve(this->queue.size());
416 while (!this->queue.empty()) {
417 tmp.push_back(this->queue.front());
418 this->queue.pop();
419 }
420 return std::move(tmp);
421 }
