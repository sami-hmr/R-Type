1834 // HOOK SYSTEM

1836
1837 // Hooks enable components to expose fields for runtime reflection and data
1838 // binding. This powers JSON configuration, network synchronization, and
1839 // dynamic component-to-component data flow.
1840 //
1841 // Hook Registration:
1842 // Components declare hookable fields using HOOKABLE() macro:
1843 // HOOKABLE(Position, HOOK(x), HOOK(y))
1844 //
1845 // This creates a static hook_map() returning field accessors.
1846 //
1847 // Hook Usage:
1848 // 1. register_hook<Component>("name", entity) - Make component fields
1849 // accessible
1850 // 2. get_hooked_value<T>("name", "field") - Read field value by string name
1851 //
1852 // Hook Applications:
1853 // - JSON configuration: "%Player:maxSpeed" reads Player component's maxSpeed
1854 // - Dynamic bindings: "#Leader:pos" creates live binding to Leader's position
1855 // - Network sync: Serialize/deserialize specific fields by name
1856 // - Runtime introspection: Query component state without compile-time
1857 // knowledge
1858 //
1859 // Hook Naming:
1860 // The "name" parameter is a user-defined identifier (often entity name or
1861 // role). Multiple entities can have hooks with different names pointing to
1862 // same component type.
1863 //
1864 // register_hook<Position>("player1", player1_entity)
1865 // register_hook<Position>("player2", player2_entity)
1866 // auto p1_x = get_hooked_value<float>("player1", "x")
