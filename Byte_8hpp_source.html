187template<typename T>
188ByteArray type_to_byte(T v, std::endian endian = std::endian::big)
189{
190 ByteArray tmp(sizeof(T));
191 auto raw = std::bit_cast<std::array<unsigned char, sizeof(T)>>(v);
192 std::copy(raw.begin(), raw.end(), tmp.begin());
193
194 if (endian != std::endian::native) {
195 std::reverse(tmp.begin(), tmp.end());
196 }
197 return tmp;
198}
