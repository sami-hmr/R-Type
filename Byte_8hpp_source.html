193template<typename T>
194ByteArray type_to_byte(T v, std::endian endian = std::endian::big)
195{
196 static_assert(sizeof(T) <= 64, "Type too large for type_to_byte");
197 ByteArray tmp(sizeof(T));
198 unsigned char* ptr = reinterpret_cast<unsigned char*>(&v);
199 std::copy(ptr, ptr + sizeof(T), tmp.begin());
200
201 if (endian != std::endian::native) {
202 std::reverse(tmp.begin(), tmp.end());
203 }
204 return tmp;
205}
